180  // 評価値を更新
181  if (board.nowTurn == player && alpha < val) {
182  alpha = val;
183  if (depth == maxDepth)bestPut = expandPut[i];
184  }else if (board.nowTurn != player && beta > val) {
185  beta = val;

188  }

199  // ボードを引数とし，合法手ボードを取得
200  unsigned long long legalBoard = makeLegalBoard(board.playerBoard, board.opponentBoard);
201  int nodeNum = bitCount(legalBoard);
