256 unsigned long long makeLegalBoard(unsigned long long playerBoard, unsigned long long opponentBoard) {
257  // 左右端，上下端，全辺と対戦相手ボードの積を取る
258  unsigned long long horWatchBoard = opponentBoard & 0x7e7e7e7e7e7e7e7e;
259  unsigned long long verWatchBoard = opponentBoard & 0x00FFFFFFFFFFFF00;
260  unsigned long long allSideWatchBoard = opponentBoard & 0x007e7e7e7e7e7e00;
261 
262  unsigned long long blankBoard = ~(playerBoard | opponentBoard); // 空いているマス
263  unsigned long long tmp; // 隣に相手の石があるか保存する
264  unsigned long long legalBoard; // 合法手ボード
265 
266 
267  // 自分の石を一方向にずらし，相手の石があれば保存
268  // 保存したものを更に5回ずらして，連続して隣接した石があれば保存していく
269  // 自分の石から連続して置かれた相手石の更に1つ隣が空いていれば置ける場所とする
270  // なお，左右ずらしだと左右端，上下ずらしだと上下端，斜めずらしだと全辺と，
271  // 先ほど作った端ボードが邪魔をすることで場外置きを防止している
272 
273  // 左
274  tmp = horWatchBoard & (playerBoard << 1);
275  for (int i = 0; i < 5; i++) tmp |= horWatchBoard & (tmp << 1);
276  legalBoard = blankBoard & (tmp << 1);
277  // 右
278  tmp = horWatchBoard & (playerBoard >> 1);
279  for (int i = 0; i < 5; i++) tmp |= horWatchBoard & (tmp >> 1);
280  legalBoard |= blankBoard & (tmp >> 1);
281  // 上
282  tmp = verWatchBoard & (playerBoard << 8);
283  for (int i = 0; i < 5; i++) tmp |= verWatchBoard & (tmp << 8);
284  legalBoard |= blankBoard & (tmp << 8);
285  // 下
286  tmp = verWatchBoard & (playerBoard >> 8);
287  for (int i = 0; i < 5; i++) tmp |= verWatchBoard & (tmp >> 8);
288  legalBoard |= blankBoard & (tmp >> 8);
289  // 右上
290  tmp = allSideWatchBoard & (playerBoard << 7);
291  for (int i = 0; i < 5; i++) tmp |= allSideWatchBoard & (tmp << 7);
292  legalBoard |= blankBoard & (tmp << 7);
293  // 左上
294  tmp = allSideWatchBoard & (playerBoard << 9);
295  for (int i = 0; i < 5; i++) tmp |= allSideWatchBoard & (tmp << 9);
296  legalBoard |= blankBoard & (tmp << 9);
297  // 左下
298  tmp = allSideWatchBoard & (playerBoard >> 9);
299  for (int i = 0; i < 5; i++) tmp |= allSideWatchBoard & (tmp >> 9);
300  legalBoard |= blankBoard & (tmp >> 9);
301  // 右下
302  tmp = allSideWatchBoard & (playerBoard >> 7);
303  for (int i = 0; i < 5; i++) tmp |= allSideWatchBoard & (tmp >> 7);
304  legalBoard |= blankBoard & (tmp >> 7);

306  return legalBoard;
